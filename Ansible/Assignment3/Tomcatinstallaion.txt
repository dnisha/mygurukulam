sudo groupadd --system tomcat
sudo useradd -r -d "/home/tomcat" -s /bin/bash tomcat -g tomcat

wget https://archive.apache.org/dist/tomcat/tomcat-7/v7.0.108/src/apache-tomcat-7.0.108-src.tar.gz
tar -xzvf apache-tomcat-7.*.tar.gz

cd apache-tomcat-7.0.108-src
sudo cp -r * /opt/tomcat7/

sudo chown -R tomcat: /opt/tomcat7

sudo nano /etc/systemd/system/tomcat.service

sudo systemctl daemon-reload

sudo chmod +x /opt/tomcat7/bin/catalina.sh 
sudo chmod +x /opt/tomcat7/bin/startup.sh

sudo mkdir /opt/tomcat7/logs
sudo chown tomcat:tomcat logs

sudo mkdir /opt/tomcat7/temp
sudo chown tomcat:tomcat temp

sudo chmod +x shutdown.sh 
sudo chown tomcat:tomcat shutdown.sh
